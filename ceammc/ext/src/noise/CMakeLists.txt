set(NOISE_SOURCES "")
macro(ceammc_noise_extension name)
    list(APPEND NOISE_SOURCES "noise_${name}.c")
endmacro()

macro(ceammc_faust_noise_extension name)
    list(APPEND NOISE_SOURCES "noise_${name}.cpp")
    ceammc_faust_gen(noise ${name})
endmacro()
